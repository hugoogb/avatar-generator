To start using `@avatar-generator` in your project, install the framework-specific package you need. Choose your preferred package manager for installation instructions.

### React

<Tabs syncKey="package-manager">
  <TabItem id="npm" label="npm">
    <Code code={`npm install @avatar-generator/react`} lang="sh"></Code>
  </TabItem>
  <TabItem id="pnpm" label="pnpm">
    <Code code={`pnpm add @avatar-generator/react`} lang="sh"></Code>
  </TabItem>
  <TabItem id="yarn" label="yarn">
    <Code code={`yarn add @avatar-generator/react`} lang="sh"></Code>
  </TabItem>
</Tabs>
